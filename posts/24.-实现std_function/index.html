#include <iostream>
#include <functional>

int myprint()
{
    std::cout << __func__ << std::endl;
    return 1;
}

namespace Mico {
    template<typename RetType, typename... Args>
    class Functional {};

    template<typename RetType, typename... Args>
    class Functional<RetType(Args...)> {
        using FuncPtrType = RetType (*)(Args...);
    public:
        Functional(FuncPtrType func) : _func(func)
        {

        }

        RetType operator()(Args... args) 
        {   
            std::cout << __func__ << std::endl;
            return _func(args...);
        }

    private:
        FuncPtrType _func;
    };
}


int main ()
{
    Mico::Functional<int()> fp{myprint};
    fp();
}

    template<typename RetType, typename... Args>
    class Functional {};

    // 偏特化
    template<typename RetType, typename... Args>
    class Functional<RetType(Args...)> {
        using FuncPtrType = RetType (*)(Args...);

    class ICallable {
    public:
        virtual ~ICallable() = default;
        virtual RetType Invoke(Args...) = 0;
    };

    template <typename T>
    class CallableT : public ICallable {
    public:
        CallableT(const T& t)
            : t_(t) {
        }

        ~CallableT() override = default;

        RetType Invoke(Args... args) override {
            return t_(args...);
        }

    private:
        T t_;
    };

    public:
        template <typename T>
        Functional& operator=(T t) {
            callable_ = std::make_unique<CallableT<T>>(t);
            return *this;
        }

        RetType operator()(Args... args) 
        {   
            std::cout << __func__ << std::endl;
            return callable_->Invoke(args...);
        }

    private:
        std::unique_ptr<ICallable> callable_;
    };
