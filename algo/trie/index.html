#include <iostream>
#include <stdio.h>
using namespace std;

const int N = 100010;
// 子节点编号(索引)
int son[N][26];
int cnt[N];
// 当前使用节点的索引
// 索引为0的点既是根节点 也是空节点
int idx; 

int insert(const char* s)
{
    int p = 0;
    for (int i = 0; s[i]; i++) {
        int u = s[i] - 'a';
        if (!son[p][u]) son[p][u] = ++idx;
        p = son[p][u];
    }
    cnt[p] ++;
}

int query(const char *s)
{
    int p = 0;
    for (int i = 0; s[i]; i++) {
        int u = s[i] - 'a';
        if (!son[p][u]) return 0;
        p = son[p][u];
    }
    return cnt[p];
}

char str[N];
int main ()
{
    int n = 0;
    char op;
    scanf("%d\n", &n);

    while(n--) {
        scanf("%c %s\n", &op, &str[0]);
        if (op == 'I') insert(str);
        else printf("%d\n", query(str));
    }
    
}
