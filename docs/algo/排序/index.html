快排

void quick_sort(int *q, int l, int r)
{
    // 边界处理, 记住其中一种即可
    // 取端点可能会退化为O(n)
    // 左端点要用 (l,j), (j+1,r)
    // int x = q[l];
    // 右端点要用 (l,i-1), (i,r)
    // int x = q[r];
    // int x = q[(l+r+1) >> 1];
    int x = q[rand() % (r - l + 1) + l];

    // off-the-end, 后面直接先增加
    int i = l - 1;
    int j = r + 1;
    
    if (l >= r) return;
    
    while (i < j) {
        do i++; while (q[i] < x);
        do j--; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    quick_sort(q, l, j);
    quick_sort(q, j+1, r);
}
